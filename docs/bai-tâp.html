Ví dụ Kết quả

Lời giải 

def parseString(l):
    items = l.split(',')
    answer = []
    for i in range(len(items)):
        item = items[i].replace('[','').replace(']','').strip() 
        item = int(item)
        answer.append(item)
    return answer
